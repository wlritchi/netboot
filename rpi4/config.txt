# Raspberry Pi 4 boot configuration for UEFI netboot
# This config tells the Pi to boot in 64-bit mode and use RPI_EFI.fd as the firmware

# Enable 64-bit mode
arm_64bit=1

# Use RPI_EFI.fd as the UEFI firmware stub
armstub=RPI_EFI.fd

# Enable UART for debugging (optional)
enable_uart=1

# No boot delay
boot_delay=0
